<?php

global $get_field;

if(!isset($get_field) || empty($get_field)) {
    $get_field = 'get_field';
}

$context['title'] = $get_field('content_faqs_title');
$context['description'] = $get_field('content_faqs_description');
$context['faqs'] = $get_field('content_faqs');

if(!empty($context['faqs'])) {
    Timber::render('tp-content_faqs.twig', $context);
}
                                      

<!-- CONTENT FAQS -->
<section class="section-default section-professional" data-aos="fade-up">
    <div class="container">
        <div class="row justify-content-center justify-content-lg-between">
            <div class="col-10 col-md-9 col-lg-5 mb-4">
                {% if title is not empty %}
                    <h2 class="h1">{{ title }}</h2>
                {% endif %}
                {% if description is not empty %}
                    <h3 class="sh1">{{ description }}</h3>
                {% endif %}
            </div>
            <div class="col-10 col-md-9 col-lg-6">
                <div class="accordion" id="faq_accordion">
                    {% set iAux = 0 %}
                    {% for faq in faqs %}
                        {% set iAux = iAux + 1 %}
                        <div id="heading_{{ iAux }}">
                            <button class="{{ (iAux != 1) ? 'collapsed' : '' }}" type="button" data-toggle="collapse" data-target="#faq_{{ iAux }}" aria-expanded="true" aria-controls="faq_{{ iAux }}">
                                <h4 class="collapse-title h4">{{ faq.question }}</h4>
                            </button>
                        </div>

                        <div id="faq_{{ iAux }}" class="collapse {{ (iAux == 1) ? 'show' : '' }}" aria-labelledby="heading_{{ iAux }}" data-parent="#faq_accordion">
                            <div class="collapse-body">
                                <p>{{ faq.answer }}</p>
                            </div>
                        </div>
                        <hr class="hr-dashed">
                    {% endfor %}
                </div>
            </div>
        </div>
    </div>
</section>
